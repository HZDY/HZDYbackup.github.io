公式本体

sin⁡2α=2sin⁡αcos⁡α\displaystyle \sin 2\alpha =2\sin \alpha \cos \alpha sin2α=2sinαcosα

cos⁡2α=cos⁡2α−sin⁡2α=2cos⁡2α−1=1−2sin⁡2α\displaystyle \cos 2\alpha =\cos^{2} \alpha -\sin^{2} \alpha =2\cos^{2} \alpha -1=1-2\sin^{2} \alpha cos2α=cos2α−sin2α=2cos2α−1=1−2sin2α

tan⁡2α=2tan⁡α1−tan⁡2α\displaystyle \tan 2\alpha =\frac{2\tan \alpha }{1-\tan^{2} \alpha }tan2α=1−tan2α2tanα​

cos⁡2α=cos⁡2α−12\displaystyle \cos^{2} \alpha =\frac{\cos 2\alpha -1}{2}cos2α=2cos2α−1​

cos⁡α+cos⁡β=2cos⁡α+β2cos⁡α−β2\displaystyle \cos \alpha +\cos \beta =2\cos\frac{\alpha +\beta }{2}\cos\frac{\alpha -\beta }{2}cosα+cosβ=2cos2α+β​cos2α−β​

tan⁡α±tan⁡β=sin⁡(α±β)cos⁡αcos⁡β\displaystyle \tan \alpha \pm \tan \beta =\frac{\sin( \alpha \pm \beta )}{\cos \alpha \cos \beta }tanα±tanβ=cosαcosβsin(α±β)​

cos⁡αcos⁡β=12[cos⁡(α−β)+cos⁡(α+β)]\cos \alpha \cos \beta =\frac{1}{2} [\cos (\alpha -\beta )+\cos (\alpha +\beta )]cosαcosβ=21​[cos(α−β)+cos(α+β)]
